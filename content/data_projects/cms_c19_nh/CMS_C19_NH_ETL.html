The CMS data was downloaded as separate files for a total compressed size of over 10GB. Each file contained one or more CSV documents with a compression ratio 9:1 (90%). The following code iterates over the compressed files, unzips them, and converts the CSV files to parquet format resulting in a total uncompressed size of just over 9GB. The parquet files are loaded into to year/month folder structure on disk.

Unzip and convert CSV files to parquet (long run time) 
# `obs_data_files` is a list of files in `data_dir` that match the pattern "zip"
# `queue` is a list of source files grouped by destination paths in `data_dir` 
queue <- { define(
  data.table()
  # Retrieve the paths of the source files
  , from = dir(
      path = getOption("params")$ext_data_dir |> paste0("/data files")
      , pattern = "zip"
      , recursive = TRUE
      , full.names = TRUE
      )
  # Extract the date from the file path
  , date_str = stri_extract_last_regex(from, "([0-9]{2}[[:punct:][[:space:]]]){2}[0-9]{2,4}") |>
      stri_replace_all_regex("[[:punct:][[:space:]]]", "/", vectorize_all = FALSE)
  # Determine the date format
  , fmts = map(date_str, \(i) lubridate::guess_formats(i, orders = c("mdy", "dmy", "ymd")) |> first())
  # Create the destination paths
  , to = pmap_chr(list(from, date_str, fmts), \(path, date_str, fmt){ 
      sprintf(
        fmt = "%s/parquet/%.0f/%s"
        , dirname(path) |> dirname()
        , lubridate::parse_date_time(date_str, orders = fmt) |> lubridate::year()
        , lubridate::parse_date_time(date_str, orders = fmt) |> lubridate::month() |> stri_pad_left(width = 2, pad = "0")
        )
    })
  , clean = !duplicated(to)
  # Remove unneeded columns
  , ~from + to + clean
  , .SD[order(to, !clean, from)]
  )[] |>
  array_branch(margin = 1)
}

# If there are no parquet files in '`data_dir`/parquet', 
# unzip the files in `obs_data_files` and create container directories. 
# Files are in CSV format and will be converted to parquet format in parallel.
if (!rlang::is_empty(queue) && {
  reduce(queue, rbind)[, 2] |> 
    unique() |> 
    map_lgl(\(i) dir(i, pattern = "parquet") |> rlang::is_empty()) |> 
    any()
  }){
  # Split the list of files in `data_dir` into n chunks for parallel processing
  cl <- parallelly::makeClusterPSOCK(4, autoStop = TRUE);
  
  # Export object `unzip.iter` to each cluster node
  parallel::clusterExport(cl = cl, varlist = c("unzip.iter"));
  
  tictoc::tic(glue::glue("[{Sys.time()}] Process Compressed Files"));
  process_log <- { 
    parallel::parLapplyLB(cl = cl, X = queue, \(x){
      out <- tryCatch(
        unzip.iter(
          from = x[1]
           , to = x[2]
           , clean = x[3] |> trimws() |> as.logical()
           )
        , error = \(e){ 
            message(glue::glue("Error: {e} <{x[1]}>"), sep = "\n");
            return(FALSE);
          }
        );
      
      gc();
      out
    })
  }
  tictoc::toc(log = TRUE);
  
  gc();
  parallel::stopCluster(cl = cl)
  
  # Review potential errors ----
    queue[map_lgl(process_log, \(p) rlang::`%||%`(p, FALSE) == FALSE )] |> 
    reduce(rbind) |> 
    View()
  
  rm(queue, unzip.iter, cl);
}
